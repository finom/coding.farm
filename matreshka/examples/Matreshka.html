Matreshka Examples

Element binding

				var mk = new MK(); // same as new Matreshka;

				mk.bindElement( 'x', '.my-select', {
					on: 'change',
					getValue: function() {
						return this.value;
					},
					setValue: function( v ) {
						this.value = v;
					}
				});

				mk.x = 3;

Element binding using predefined Element Options

					var mk = new MK(); // same as new Matreshka;
					mk.bindElement( 'x', '.my-select' );
					mk.x = 3;

Binding custom element

					// custom checkbox behavior
					$( document ).on( 'click',  '.custom-checkbox', function() {
						$( this ).toggleClass( 'checked' );
					});
					
				
					var mk = new MK();

					mk.bindElement( 'x', '.my-checkbox', {
						on: 'click',
						getValue: function() {
							return $( this ).hasClass( 'checked' );
						},
						setValue: function( v ) {
							$( this ).toggleClass( 'checked', !!v );
						}
					});

					mk.x = true;
					
				
					
					.custom-checkbox {
						background-image: url(...);
						position: relative;
						display: inline-block;
						width: 14px;
						height: 14px;
					}

					.custom-checkbox.checked {
						background-position: -14px;
					}

Events

					var mk = new MK(); // same as new Matreshka;

					mk.on( 'yeah', function() {
						var args = [].slice.call( arguments ).join( ', ' );
						$( '.output' ).html( args );
					});

					mk.trigger( 'yeah', 1, 'string', {} );

change: event

					var mk = new MK();

					mk.on( 'change:x', function() {
						var text = 'x changed to ' + this.x;
						$( '.output' ).html( text );
					});

					mk.x = 5;

Inheritance

					var MyApp = Class({
						'extends': MK,
						constructor: function() {
							this
								//calling this method is required before you use events and bindings
								.initMK()
								// default data
								.set( 'x', 3 )
								.bindElement( 'x', '.my-select' )
								.on( 'change:x', function() {
									alert( 'x changed to ' + this.x );
								})
							;
						}
					});

					var app = new MyApp();
