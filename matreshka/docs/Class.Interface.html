Class: Interface

Class. Interface

new Interface(parent)

Simple interface implementation

Btw Class function supports interfaces. You can create your own Interface constructor that should have .validate method and use it same way as described in examples. Pass interface instance to 'implements' property.
Arguments
parent <optional>
 		Interface parent interface
Array string interface properties

Examples

Basic usage

var myInterface = new Class.Interface( 'method1', 'method2' );
var MyClass({
 'implements': myInterface,
	method1: function() { ... },
	method2: function() { ... }
});

Method is not implemented in class (error)

var myInterface = new Class.Interface( 'method1', 'method2' );
var MyClass({
 'implements': myInterface,
	method1: function() { ... },
});

Inheritage

var interface1 = new Class.Interface( 'method1', 'method2' );
var interface2 = new Class.Interface( interface1, 'method3' );
var MyClass({
 'implements': interface2,
	method1: function() { ... },
	method2: function() { ... },
	method3: function() { ... },
});
